    SpelloutRules {
        // 1 is "eins" when by itself, but turns into "ein" in most

        "    -x: minus >>;\n"
        "    x.x: << komma >>;\n"

        "%%main:\n"
               // words for numbers from 0 to 12.  Notice that the values
               // from 13 to 19 can derived algorithmically, unlike in most
               // other languages

               // goes on the front
